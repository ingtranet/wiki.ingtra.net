export function EasyTable({ text }) {
  const rows = text.split('\n').filter(s => s.trim())
  return (
    <table>
      <tbody>
        {
          rows.map(
            row => (
              <tr>
                {row.split('    ').map(el => (<td>{el}</td>))}
              </tr>
            )
          )
        }
      </tbody>
    </table>
  )
}
