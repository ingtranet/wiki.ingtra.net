import IpSubnetCalculator from 'ip-subnet-calculator'

export const raw = [
  {cidr: '10.0.0.0/24',     vlan: ''     , description: 'Gateway & Temporary zone for newly added'},
  {cidr: '10.0.1.0/24',     vlan: '1'    , description: 'Network Components (ex. DNS)'},
  {cidr: '10.0.16.0/24',    vlan: '16'   , description: 'Ubiquiti Network Devices'},
  {cidr: '10.0.80.0/24',    vlan: '80'   , description: 'Internal Network for Proxmox Hosts'},
  {cidr: '10.1.0.0/24',     vlan: '100'  , description: 'Residents Users'},
  {cidr: '10.1.1.0/24',     vlan: '101'  , description: 'Nya Project Users'},
  {cidr: '10.1.2.0/24',     vlan: '102'  , description: 'Multimedia Devices'},
  {cidr: '10.1.3.0/24',     vlan: '103'  , description: 'Network Components (ex. DNS)'},
  {cidr: '10.1.4.0/24',     vlan: '104'  , description: 'Guest Users'},
  {cidr: '10.1.5.0/24',     vlan: '105'  , description: 'IoT Devices'},
  
]

export const data = raw.map(item => {
    const ip = item.cidr.split('/')[0]
    const prefix = item.cidr.split('/')[1]
    const calc = IpSubnetCalculator.calculateSubnetMask(ip, prefix)
    return {
      cidr: item.cidr,
      start: calc.ipLowStr,
      end: calc.ipHighStr,
      mask: calc.prefixMaskStr,
      vlan: item.vlan,
      description: item.description
    }
  }
)
